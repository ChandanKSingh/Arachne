115 class SpinLock {
116  public:
118  explicit SpinLock(std::string name) : state(false), name(name) {}
119  SpinLock() : state(false), name("unnamed") {}
120  ~SpinLock(){}
121 
123  inline void
124  lock() {
125  uint64_t startOfContention = 0;
126  while (state.exchange(true, std::memory_order_acquire) != false) {
127  if (startOfContention == 0) {
128  startOfContention = Cycles::rdtsc();
129  } else {
130  uint64_t now = Cycles::rdtsc();
131  if (Cycles::toSeconds(now - startOfContention) > 1.0) {
132  fprintf(errorStream,
133  "%s SpinLock locked for one second; deadlock?\n",
134  name.c_str());
135  startOfContention = now;
136  }
137  }
138  yield();
139  }
140  }
141 
147  inline bool
148  try_lock() {
149  // If the original value was false, then we successfully acquired the
150  // lock. Otherwise we failed.
151  return !state.exchange(true, std::memory_order_acquire);
152  }
153 
155  inline void
156  unlock() {
157  state.store(false, std::memory_order_release);
158  }
159 
161  inline void
162  setName(std::string name) {
163  this->name = name;
164  }
165 
166  private:
167  // Implements the lock: false means free, true means locked
168  std::atomic<bool> state;
169 
170  // Descriptive name for this SpinLock. Used to identify the purpose of
171  // the lock, what it protects, where it exists in the codebase, etc.
172  //
173  // Used to identify the lock when reporting a potential deadlock.
174  std::string name;
175 };

415 template<typename _Callable, typename... _Args>
416 ThreadId
417 createThread(int kId, _Callable&& __f, _Args&&... __args) {
418  if (kId == -1)
419  kId = kernelThreadId;
420 
421  auto task = std::bind(
422  std::forward<_Callable>(__f), std::forward<_Args>(__args)...);
423 
424  bool success;
425  uint32_t index;
426  do {
427  // Each iteration through this loop makes one attempt to enqueue the
428  // task to the specified core. Multiple iterations are required only if
429  // there is contention for the core's state variables.
430  MaskAndCount slotMap = *occupiedAndCount[kId];
431  MaskAndCount oldSlotMap = slotMap;
432 
433  if (slotMap.numOccupied >= maxThreadsPerCore)
434  return NullThread;
435 
436  // Search for a non-occupied slot and attempt to reserve the slot
437  index = 0;
438  while ((slotMap.occupied & (1L << index)) && index < maxThreadsPerCore)
439  index++;
440 
441  slotMap.occupied =
442  (slotMap.occupied | (1L << index)) & 0x00FFFFFFFFFFFFFF;
443  slotMap.numOccupied++;
444 
445  success = occupiedAndCount[kId]->compare_exchange_strong(oldSlotMap,
446  slotMap);
447  } while (!success);
448 
449  // Copy the thread invocation into the byte array.
450  new (&allThreadContexts[kId][index]->threadInvocation)
451  Arachne::ThreadInvocation<decltype(task)>(task);
452  allThreadContexts[kId][index]->wakeupTimeInCycles = 0;
453  return ThreadId(allThreadContexts[kId][index],
454  allThreadContexts[kId][index]->generation);
455 }

477 template<typename _Callable, typename... _Args>
478 ThreadId
479 createThread(_Callable&& __f, _Args&&... __args) {
480  // Find a kernel thread to enqueue to by picking two at random and choosing
481  // the one with the fewest Arachne threads.
482  int kId;
483  int choice1 = static_cast<int>(random()) % numCores;
484  int choice2 = static_cast<int>(random()) % numCores;
485  while (choice2 == choice1 && numCores > 1)
486  choice2 = static_cast<int>(random()) % numCores;
487 
488  if (occupiedAndCount[choice1]->load().numOccupied <
489  occupiedAndCount[choice2]->load().numOccupied)
490  kId = choice1;
491  else
492  kId = choice2;
493 
494  return createThread(kId, __f, __args...);
495 }
